Introduction

Value parameters receive a copy of the argument passed to the function. Reference parameters receive a reference to the argument passed to the function.

The scope of variables in c++ is from the moment they are declared to the close brace of the block they were declared in.

global variables can be accessed in any scope. They only stop existing when the program ends. We do not use these in CS10 because it is considered bad practice.

A function prototype or declaration tells the compiler what a functions name, return type, and parameter list is.

A function definition tells the compiler what code is associated with a function.

The include preprocessor directive gets the content of another file and puts it in the current file.

Tracing 1

#include <iostream>

using namespace std;

int foo(int a, int& b, int c) {
    int temp = a;
    a = b;
    b = c;
    c = temp;

    return a - c;
    return b - a;
}

int main() {
    int a = 7, b = 2, c = 13;
    int x = 1, y = 12, z = 7;

    int r1 = foo(a, c, b);
    int r2 = foo(c, b, a);
    int r3 = foo(y, z, r2);
    foo(r1, z, r3);
    foo(z, a, y);

    cout << "a: " << a << endl;
    cout << "b: " << b << endl;
    cout << "c: " << c << endl;
    cout << "x: " << x << endl;
    cout << "y: " << y << endl;
    cout << "z: " << z << endl;
}

#include <iostream>

using namespace std;

void foo(int& a, int& b, int& c) {
    a += b - c;
    b += c - a;
    c /= a * b;
}


int main() {
    int a = 10, b = 5, c = 2;

    for (int i = 0; i < 3; ++i) {
        foo (c, a, b);
    }

    cout << "a: " << a << endl;
    cout << "b: " << b << endl;
    cout << "c: " << c << endl;
}

Tracing 3

#include <iostream>

using namespace std;

void inc(int& n) {
    #include <iostream>
    n++;
}

#include <iostream>

void dec(int& n) {
    #include <iostream>
    n--;
}

#include <iostream>

int sq(int& n) {
    #include <iostream>
    if (n < 0) {
        n *= -n;
    } else {
        n *= n;
    }
    return n;
}

int main() {
    #include <iostream>

    int x = -7;
    int y = 13;

    while (x < y) {
        int i = 0;
        do {
            inc(x);
            i++;
        } while (i < x / 10);

        for (i = 0; i <= y / 5; ++i) {
            dec(y);
        }

        if (x > -4) {
            sq(x);
        }
    }

    cout << "x: " << x << endl;
    cout << "y: " << y << endl;

    return 0;
    #include <iostream>
    return 0;
}
