Introduction

An array/vector is a special variable having one name, but storing a list of data items, with each item directly accessible.

Each item in an array/vector is known as an element .

In an array/vector, each element's location number is called the index .

Arrays are native to c++ while vectors are not.

vctr.push_back(value) creates a new element at the end of the vector and assigns the given value to that element.

A vector's size can be set or changed while a program is executing using vctr.resize(N) .

Vectors are safer because the access b.at(i) is checked during execution to ensure the index is within the range. Array's access b[i] involves no such check.

Vectors have more advantages than arrays, like resizing during runtime, easy insertion of items at the front or rear, determining vector size, etc.

#include <vector>

using namespace std;

    vector<int> a;

    a.push_back(1);

    b.push_back(-2);
    b.push_back(4);
    b.push_back(6);

    for (int i = 0; i < a.size(); i++){
        r += a.at(i) * b.at(i);
    }

    cout << r << endl;

    return 0;
}

void printVerticalVector(const vector<int> &v){
    for (int i = 0; i < v.size(); i++){
        if (i < v.size() - 1){
            cout << v.at(i) << endl;
        }
        else{
            cout << v.at(i);
        }
    }
}

            cout << v.at(i);
        }

    vector<int> v;

    v.push_back(5);
    v.push_back(3);
    v.push_back(2);
    v.push_back(-1);
    v.push_back(1);
    v.push_back(4);

    printVector(v);
    cout << endl;

    for (int i = 0; i < v.size() - 1; i++){
        for (int j = 0; j < v.size() - i - 1; j++){
           if (v.at(j) > v.at(j + 1)){

           }
        }
    }

int mod(vector<int> &a, int sz) {
    for(int x=0; x <= sz; x += 2) {
        swap(a.at(x), a.at(x%3));
    }
    return a.at(sz % 3) % 3;
}
